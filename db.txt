Table users as u {
  id serial [pk, increment]
  created_at timestamp
  username varchar(15)
  updated_at timestamp
}

Table comments as c {
  comment_id serial  [pk, increment]
  comment varchar(240)
  post_id int
  user_id int 
  created_at timestamp
}

Table posts as p {
  id serial [pk, increment]
  post varchar
  url varchar(200)
  user_id int 
  created_at timestamp
  updated_at timestamp
  lat REAL 
  lng REAL 
}

Table likes as l {
  id serial [pk, increment]
  user_id int
  post_id int 
  comment_id int
}

Table photo_tags as pt {
  id serial [pk, increment]
  created_at timestamp
  updated_at timestamp 
  user_id int [ref: > u.id ]
  post_id int [ref: > p.id]
  x_pos int 
  y_pos int 
}

Table caption_tag as ct {
  id serial [pk,increment]
  created_at timestamp
  post_id int [ref: > p.id]
  user_id int [ref: > u.id]
}

Ref: u.id < c.user_id
Ref: c.post_id > p.id
Ref: u.id < p.user_id
Ref: u.id < l.user_id
Ref: p.id - l.post_id
Ref: l.comment_id > c.comment_id
